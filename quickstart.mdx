## Get started in five steps

Convert any document into high quality, LLM ready data.


### Step 1: Install the Ingestor SDK 

<CodeGroup>
  ```bash Python
  pip install ingestorai
  ```

  ```bash Node
  npm install ingestorai
  # or: pnpm add ingestorai
  ```
</CodeGroup>


### Step 2: Initialize client

Navigate to your account dashboard to obtain a copy of your API key

<CodeGroup>
  ```bash Python
  from ingestor import Ingestor

  client = Ingestor(api_key="INGESTOR_API_KEY")
  ```

  ```bash Node
  import { Ingestor } from "ingestorai";

  const client = new Ingestor({ apiKey: "INGESTOR_API_KEY" });
  ```
</CodeGroup>


### Step 3: Call the document parsing API

<CodeGroup>
  ```bash Python
  # input formats: email, pdf, docx, doc, xlsx, xls, csv, xml, jpg, jpeg, png, gif, bmp, tiff, pptx, ppt
  # input can be a local path or file-like object

  response = client.content.parse(
      input="loan_application_john_smith.pdf",  
      processing_options={
          "ocr_mode": "layout_aware",
          "classify_content": ["bank_statement", "loan_application", "id_document", "other"], 
          "generate_document_title_metadescription": True,
          "split_into_documents": True, # If your file contains multiple documents (e.g. a scan of several forms).
          "chunking_strategy": "page",
          "contextualize_chunks": True, # Agentic process adds missing context            
          "generate_table_summary": True, 
          "generate_figure_summary": True
      }
  )

  job_id = response["job_id"]
  ```

  ```bash Node
  // input formats: email, pdf, docx, doc, xlsx, xls, csv, xml, jpg, jpeg, png, gif, bmp, tiff, pptx, ppt
  // input can be a local path, File/Blob (browser), or Buffer/stream

  const response = await client.content.parse({
    input: "loan_application_john_smith.pdf",
    processing_options: {
      ocr_mode: "layout_aware",
      classify_content: ["bank_statement", "loan_application", "id_document", "other"],
      generate_document_title_metadescription: true,
      split_into_documents: true,
      chunking_strategy: "page",
      contextualize_chunks: true,
      generate_table_summary: true,
      generate_figure_summary: true
    }
  });

  const job_id = response.job_id;
  ```
</CodeGroup>

### Step 4: Poll for job completion

<Info>
Because jobs run through a multi-step pipeline, they may take several minutes to complete. Additional `processing_options` can further increase this time.
</Info>

<CodeGroup>
  ```bash Python
  documents = client.jobs.get_document_response(job_id, include=["layout_bbox", "blocks", "page_images"])
  print(documents)
  ```

  ```bash Node
  const documents = await client.jobs.get_document_response(job_id, { include: ["layout_bbox", "blocks"] });
  console.log(documents);
  ```
</CodeGroup>

### Step 5: Extract key-value fields (optional)

Use our SDK to extract specific key-value pairs from the parsed document response. 
This feature is free to use, but requires your own `OPENAI_API_KEY`.

<CodeGroup>
    ```bash Python
  import os
  from pydantic import BaseModel, Field

  OPENAI_KEY = os.getenv("OPENAI_API_KEY")

  class BankStatement(BaseModel):
      account_holder: str = Field(..., description="First and last name")
      closing_balance: float
      statement_period_start: str = Field(..., description="ISO date")
      statement_period_end: str = Field(..., description="ISO date")

  for doc in documents["result"]:
      if doc.get("classification") == "bank_statement":
          extracted = client.content.extract(
              model="gpt-5-mini",
              openai_api_key=OPENAI_KEY,
              parsed_document=doc,   
              extraction_schema=BankStatement,
          )
          print(extracted)
  ```

  ```bash Node
  const { z } = require("zod");
  
  const OPENAI_KEY = process.env.OPENAI_API_KEY;

  const BankStatement = z.object({
    account_holder: z.string(),
    closing_balance: z.number(),
    statement_period_start: z.string(), // ISO date
    statement_period_end: z.string(),   // ISO date
  });

  for (const doc of documents.result) {
    if (doc.classification === "bank_statement") {
      const extracted = await client.content.extract({
        model: "gpt-5-mini",
        openaiApiKey: OPENAI_KEY,
        parsedDocument: doc,
        extractionSchema: BankStatement,
      });
      console.log(extracted);
    }
  }
  ```
